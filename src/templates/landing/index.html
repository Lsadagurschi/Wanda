Sua pergunta:

"Quais foram os 5 produtos mais vendidos no último trimestre, com o total de receita de cada um?"

-- SQL gerado pela Wanda

SELECT p.nome_produto,

SUM(v.quantidade) AS total_vendido,

SUM(v.quantidade * p.preco) AS receita_total

FROM vendas v

JOIN produtos p ON v.produto_id = p.id

WHERE v.data >= DATE_SUB(NOW(), INTERVAL 3 MONTH)

GROUP BY p.nome_produto

ORDER BY receita_total DESC

LIMIT 5;
